34 // 0xxx_xxxx
35 // 110x_xxxx 10xx_xxxx
36 // 1110_xxxx 10xx_xxxx 10xx_xxxx

50  template<typename PeekFn>
51  static size_t char_size(PeekFn && peek_fn)
52  {
53  char_type const ch0 = std::forward<PeekFn>(peek_fn)();
54  if (ch0 < 0x80) // 0xxx_xxxx
55  return 1;
56  if (ch0 < 0xC0)
57  return 1;
58  //throw std::runtime_error("The utf8 first char in sequence is incorrect");
59  if (ch0 < 0xE0) // 110x_xxxx 10xx_xxxx
60  return 2;
61  if (ch0 < 0xF0) // 1110_xxxx 10xx_xxxx 10xx_xxxx
62  return 3;
63  if (ch0 < 0xF8) // 1111_0xxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
64  return 4;
65  if (ch0 < 0xFC) // 1111_10xx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
66  return 5;
67  if (ch0 < 0xFE) // 1111_110x 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
68  return 6;
69  return 1;
70  //throw std::runtime_error("The utf8 first char in sequence is incorrect");
71  }

73  template<typename ReadFn>
74  static uint32_t read(ReadFn && read_fn)
75  {
76  char_type const ch0 = read_fn();
77  if (ch0 < 0x80) // 0xxx_xxxx
78  return ch0;
79  if (ch0 < 0xC0)
80  return invalid_code_point;
81  //throw std::runtime_error("The utf8 first char in sequence is incorrect");
82  if (ch0 < 0xE0) // 110x_xxxx 10xx_xxxx
83  {
84  char_type const ch1 = read_fn(); if (ch1 >> 6 != 2) goto _err;
85  return (ch0 << 6) + ch1 - 0x3080;
86  }
87  if (ch0 < 0xF0) // 1110_xxxx 10xx_xxxx 10xx_xxxx
88  {
89  char_type const ch1 = read_fn(); if (ch1 >> 6 != 2) goto _err;
90  char_type const ch2 = read_fn(); if (ch2 >> 6 != 2) goto _err;
91  return (ch0 << 12) + (ch1 << 6) + ch2 - 0xE2080;
92  }
93  if (ch0 < 0xF8) // 1111_0xxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
94  {
95  char_type const ch1 = read_fn(); if (ch1 >> 6 != 2) goto _err;
96  char_type const ch2 = read_fn(); if (ch2 >> 6 != 2) goto _err;
97  char_type const ch3 = read_fn(); if (ch3 >> 6 != 2) goto _err;
98  return (ch0 << 18) + (ch1 << 12) + (ch2 << 6) + ch3 - 0x3C82080;
99  }
100  if (ch0 < 0xFC) // 1111_10xx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
101  {
102  char_type const ch1 = read_fn(); if (ch1 >> 6 != 2) goto _err;
103  char_type const ch2 = read_fn(); if (ch2 >> 6 != 2) goto _err;
104  char_type const ch3 = read_fn(); if (ch3 >> 6 != 2) goto _err;
105  char_type const ch4 = read_fn(); if (ch4 >> 6 != 2) goto _err;
106  return (ch0 << 24) + (ch1 << 18) + (ch2 << 12) + (ch3 << 6) + ch4 - 0xFA082080;
107  }
108  if (ch0 < 0xFE) // 1111_110x 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
109  {
110  char_type const ch1 = read_fn(); if (ch1 >> 6 != 2) goto _err;
111  char_type const ch2 = read_fn(); if (ch2 >> 6 != 2) goto _err;
112  char_type const ch3 = read_fn(); if (ch3 >> 6 != 2) goto _err;
113  char_type const ch4 = read_fn(); if (ch4 >> 6 != 2) goto _err;
114  char_type const ch5 = read_fn(); if (ch5 >> 6 != 2) goto _err;
115  return (ch0 << 30) + (ch1 << 24) + (ch2 << 18) + (ch3 << 12) + (ch4 << 6) + ch5 - 0x82082080;
116  }
117  return invalid_code_point;
118  //throw std::runtime_error("The utf8 first char in sequence is incorrect");
119  _err: return invalid_code_point;
120  //throw std::runtime_error("The utf8 slave char in sequence is incorrect");
121  }

123  template<typename WriteFn>
124  static void write(uint32_t const cp, WriteFn && write_fn)
125  {
126  if (cp < 0x80) // 0xxx_xxxx
127  write_fn(static_cast<char_type>(cp));
128  else if (cp < 0x800) // 110x_xxxx 10xx_xxxx
129  {
130  write_fn(static_cast<char_type>(0xC0 | cp >> 6));
131  goto _1;
132  }
133  else if (cp < 0x10000) // 1110_xxxx 10xx_xxxx 10xx_xxxx
134  {
135  write_fn(static_cast<char_type>(0xE0 | cp >> 12));
136  goto _2;
137  }
138  else if (cp < 0x200000) // 1111_0xxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
139  {
140  write_fn(static_cast<char_type>(0xF0 | cp >> 18));
141  goto _3;
142  }
143  else if (cp < 0x4000000) // 1111_10xx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
144  {
145  write_fn(static_cast<char_type>(0xF8 | cp >> 24));
146  goto _4;
147  }
148  else if (cp < 0x80000000) // 1111_110x 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx 10xx_xxxx
149  {
150  write_fn(static_cast<char_type>(0xFC | cp >> 30));
151  goto _5;
152  }
153  else
154  return;
155  //throw std::runtime_error("Tool large UTF8 code point");
156  return;
157  _5: write_fn(static_cast<char_type>(0x80 | (cp >> 24 & 0x3F)));
158  _4: write_fn(static_cast<char_type>(0x80 | (cp >> 18 & 0x3F)));
159  _3: write_fn(static_cast<char_type>(0x80 | (cp >> 12 & 0x3F)));
160  _2: write_fn(static_cast<char_type>(0x80 | (cp >> 6 & 0x3F)));
161  _1: write_fn(static_cast<char_type>(0x80 | (cp & 0x3F)));
162  }
163 };
