45 template<
46  typename Utf,
47  typename Outf,
48  typename It,
49  typename Oit,
50  convz_impl>
51 struct convz_strategy
52 {
53  Oit operator()(It it, Oit oit) const
54  {
55  auto const read_fn = [&it] { return *it++; };
56  auto const write_fn = [&oit] (typename Outf::char_type const ch) { *oit++ = ch; };
57  while (true)
58  {
59  auto const cp = Utf::read(read_fn);
60  if (!cp)
61  return oit;
62  Outf::write(cp, write_fn);
63  }
64  }
65 };

109 template<
110  typename Utf,
111  typename Outf,
112  typename It,
113  typename Oit,
114  conv_impl>
115 struct conv_strategy final
116 {
117  Oit operator()(It it, It const eit, Oit oit) const
118  {
119  auto const read_fn = [&it, &eit]
120  {
121  if (it == eit)
122  throw std::runtime_error("Not enough input");
123  return *it++;
124  };
125  auto const write_fn = [&oit] (typename Outf::char_type const ch) { *oit++ = ch; };
126  while (it != eit)
127  Outf::write(Utf::read(read_fn), write_fn);
128  return oit;
129  }
130 };

132 template<
133  typename Utf,
134  typename Outf,
135  typename It,
136  typename Oit>
137 struct conv_strategy<Utf, Outf, It, Oit, conv_impl::random_interator> final
138 {
139  Oit operator()(It it, It const eit, Oit oit) const
140  {
141  auto const write_fn = [&oit] (typename Outf::char_type const ch) { *oit++ = ch; };
142  if (eit - it >= static_cast<typename std::iterator_traits<It>::difference_type>(Utf::max_supported_symbol_size))
143  {
144  auto const fast_read_fn = [&it] { return *it++; };
145  auto const fast_eit = eit - Utf::max_supported_symbol_size;
146  while (it < fast_eit)
147  Outf::write(Utf::read(fast_read_fn), write_fn);
148  }
149  auto const read_fn = [&it, &eit]
150  {
151  if (it == eit)
152  throw std::runtime_error("Not enough input");
153  return *it++;
154  };
155  while (it != eit)
156  Outf::write(Utf::read(read_fn), write_fn);
157  return oit;
158  }
159 };
