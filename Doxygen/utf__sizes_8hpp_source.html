70 template<
71  typename It,
72  iterator_impl>
73 struct next_strategy final
74 {
75  void operator()(It & it, It const & eit, size_t size)
76  {
77  while (++it, --size > 0)
78  if (it == eit)
79  throw std::runtime_error("Not enough input for the forward iterator");
80  }
81 };

83 template<typename It>
84 struct next_strategy<It, iterator_impl::random_access> final
85 {
86  void operator()(It & it, It const & eit, typename std::iterator_traits<It>::difference_type const size)
87  {
88  if (eit - it < size)
89  throw std::runtime_error("Not enough input for the random access iterator");
90  it += size;
91  }
92 };

96 template<
97  typename Utf,
98  typename It,
99  typename Eit>
100 size_t size(It it, Eit const eit)
101 {
102  size_t total_cp = 0;
103  while (it != eit)
104  {
105  size_t const size = Utf::char_size([&it] { return *it; });
106  detail::next_strategy<
107  typename std::decay<It>::type,
108  std::is_base_of<std::random_access_iterator_tag, typename std::iterator_traits<typename std::decay<It>::type>::iterator_category>::value
109  ? detail::iterator_impl::random_access
110  : detail::iterator_impl::forward>()(it, eit, size);
111  ++total_cp;
112  }
113  return total_cp;
114 }
